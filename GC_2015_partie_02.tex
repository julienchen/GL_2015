\section{Introduction à UML} 
\begin{frame}
\frametitle{UML : Définition} 
\begin{itemize}
\item  Acronyme : « Unified Modeling Language »
\item  Selon l?OMG (Object Management Group), UML est un langage visuel dédié à la spécification, la construction et la documentation des artefacts d?un système.
\item  UML est le standard de facto de notation graphique pour la création et le représentation des modèles graphiques (accompagnés de texte) associés aux logiciels.
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{UML : Définition} 
\begin{itemize}
\item  UML, un ensemble de notations particulièrement vaste (contrairement au modèle Entité-Association).
\item  UML définit différents profils UML qui constituent des sous-ensembles spécialisés de la notation pour certains domaines (ex : profil EJB ? Enterprise JavaBeans - pour les diagrammes EJB).
\end{itemize}
\end{frame}



\begin{frame}
\frametitle{UML : Définition} 
\begin{itemize}

\item  Langage :
\begin{itemize}\item Syntaxe et règles d'écriture\item Notations graphiques normalisées
\end{itemize}
\item  ... de modélisation :
\begin{itemize}\item Abstraction du fonctionnement et de la structure du système\item  Spécification et conception
\end{itemize}\item  ... unifié :
\begin{itemize}\item  Fusion de plusieurs notations ante?rieures : Booch, OMT, OOSE\item  Standard défini par l'OMG (Object Management Group)\item  Dernière version : UML 2.5 (Juin 2015)
\end{itemize}
\end{itemize}
En résumé : Langage graphique pour visualiser, spécifier, construire etdocumenter un logiciel
\end{frame}



\begin{frame}
\frametitle{Pourquoi UML?} 
\begin{itemize}
\item  Besoin de modéliser pour construire un logiciel
\begin{itemize}
\item  Modélisation des aspects statiques et dynamiques\item  Modélisation à différents niveaux d'abstraction et selonplusieurs vues\item  Indépendant du processus de développement
\end{itemize}
\item Besoin de langages normalisés pour la modélisation
\begin{itemize}
 \item  Langage graphique, semi-formel\item   Standard très utilisé
 \end{itemize}
\item Conception orientée objet
\begin{itemize}\item   Façon efficace de penser le logiciel\item   Indépendance du langage de programmation (langages non objet)
 \end{itemize}
  \end{itemize}
 \end{frame}

\begin{frame}
\frametitle{Utilisation d?UML} 
Il existe trois façons d?utiliser UML?
UML en mode esquisse
\begin{itemize}
\item  Diagrammes informels et incomplets
\item  Souvent tracés à la main
\item  Créés pour expliciter des parties délicates de l?espace du problème ou de la solution en exploitant la puissance des langages graphiques.
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Utilisation d?UML} 
UML en mode plan
\begin{itemize}
\item  Diagrammes de conception relativement détaillés utiles pour :
\begin{itemize}
\item  La pro-ingénierie (génération de code à partir de diagrammes);
\item  La rétro-ingénierie, qui permet de visualiser et de mieux comprendre le code existant en générant des diagrammes UML;
\end{itemize}
\item Avant la programmation, des diagrammes détaillés peuvent servir de guide pour la génération de code, soit manuellement, soit automatiquement avec un outil.
\end{itemize}
\end{frame}



\begin{frame}
\frametitle{Utilisation d?UML} 
UML comme langage de programmation
\begin{itemize}
\item  Spécification complète et exécutable d?un système logiciel en UML. 
\item  Le code exécutable sera généré automatiquement et ne sera ni revu ni modifié par les développeurs.
\item  Toujours en développement.
\end{itemize}
\end{frame}

\section{Analyse des besoins et spécification}

\begin{frame}
\frametitle{Analyse des besoins et spécification} 
Objectif - Fournir une vue claire de ce que doit faire le logiciel : le cahier des charges fonctionnel
\begin{itemize}
\item  Définir les fonctionnalités du produit\item  Déterminer les contraintes de conception du logiciel\item  Clarifier le cahier des charges (ambiguïtés, contradictions)
\end{itemize}
 
 Enjeu - Contrat entre le client et le fournisseur :
 \begin{itemize}
\item  Pour le client : tous les besoins sont exprimés\item  Pour le fournisseur : tous les besoins exprimés sont réalisables
 \end{itemize}
 Méthode Raffinement du cahier des charges initial validé par des allers-retours avec le client
\end{frame}



 \begin{frame}
\frametitle{Spécification vs. conception} 


\begin{block}{Spécifier = Décrire le problème}
Que doit faire le logiciel ?
\end{block}

\centerline{$\Downarrow$}

\begin{alertblock}{Concevoir = Élaborer une solution}
Comment le réaliser ? 
\end{alertblock}

\end{frame}



 \begin{frame}
\frametitle{Contenu du cahier des charges fonctionnel} 

Description générale
\begin{itemize}\item Contexte d'utilisation (utilisateurs, environnement)\item Fonctionnalités\item Interface utilisateur\item Contraintes matérielles, de performance, de sécurité...
\end{itemize}
  Spécification des exigences
  \begin{itemize}\item  Exigences fonctionnelles : spécification des cas d'utilisation, scénarios d'utilisation\item Interface : prototype, storyboard, lien avec les fonctionnalités\item  Exigences non fonctionnelles : utilisabilité, performance, sécurité...
\end{itemize}
\end{frame}




 \begin{frame}
\frametitle{Contexte d'utilisation} 
Contexte d'utilisation
  \begin{itemize}\item Utilisateurs : expérience, compétences, connaissance du système\item  Utilisation : fréquence, durée d'utilisation\item  Environnement technique : matériel, ressources, réseau
  \end{itemize}Permet de spécifier les contraintes de qualité en termes de
    \begin{itemize}\item facilité d'utilisation\item  efficacité, performance\item fiabilité, portabilité
  \end{itemize}
\end{frame}




 \begin{frame}
\frametitle{Description des cas d'utilisation} 
 Objectif : Comprendre les besoins du client pour rédiger le cahier deschargesPrincipe :
   \begin{itemize}\item Définir les limites du système\item  Définir l'environnement du système : les utilisateurs ou élémentsqui interagissent avec le système\item  Définir les utilisations principales du système : à quoi sert-il ?
     \end{itemize}Éléments constitutifs :
 \begin{itemize}\item  Diagrammes des cas d'utilisation\item  Description textuelle des cas d'utilisation\item  Diagrammes de séquence des scénarios d'utilisation
\end{itemize}
\end{frame}




 \begin{frame}
\frametitle{Cas d'utilisation} 
Fonctionnalités principales du système du point de vue extérieur
Acteur : Entité qui interagit avec le système
\begin{itemize}\item   Personne, chose, logiciel, extérieur au système décrit\item   Représente un rôle (plusieurs rôles possibles pour une même entité)\item   Identifié par le nom du rôle
\end{itemize}
Cas d'utilisation : Fonctionnalité visible de l'extérieur
\begin{itemize}\item    Action déclenchée par un acteur\item    Identifié par une action (verbe à l'infinitif)
\end{itemize}

\begin{alertblock}{Résumé}
?Vision du système centrée sur l'utilisateur
\end{alertblock}

 \begin{frame}
\frametitle{Recommandations} 
Rester lisible
\begin{itemize} \item Pas plus d'une dizaine de cas dans un diagramme\item  Au besoin, faire plusieurs diagrammes (si cas disjoints entreacteurs, pour détailler un cas...)\item  Relations entre cas seulement si nécessaires et pas trop lourdes
\end{itemize}Accompagner le diagramme d'un glossaire expliquant le vocabulaire
\begin{itemize}\item  Pour chaque acteur et chaque cas d'utilisation, une phraseexpliquant le rôle de l'acteur et le scénario représenté par le cas\item  Expliquer les relations entre acteurs et entre cas (hiérarchie,extends, includes...)
\end{itemize}Pour les détails, privilégier la description textuelle des cas d'utilisation
\end{frame}


 \begin{frame}
\frametitle{Extrait d'un glossaire} 
\begin{itemize}
 \item Client : Un client est un internaute inscrit sur le site. \item Client adhérent : Un client adhérent est un client particulier inscrit au programme de fidélité du site. \item Commander : Commander un ensemble de produits proposés sur le site. Il faut posséder un compte et être identifié en tant que client pour pouvoir passer une commande. \item Gérer son compte : Regroupe les opérations de modifications de ses coordonnées et la possibilité de désactiver son compte. Il faut être identifié en tant que client pour gérer son compte. \item Utiliser points fidélité : Possibilité d'utiliser tout ou partie des pointsde fidélité acquis pendant une commande, de manière à bénéficier deréductions.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Contraintes associées au diagramme} 
Contraintes non exprimées dans le diagramme de classes
\begin{itemize}\item Le code barre d'un exemplaire et l'ISBN d'un livre sont uniques\item Un utilisateur ne peut pas emprunter plus d'un exemplaire d'un même livre\item Si un exemplaire est emprunté, sa date de retour n'est pas null\item Si un exemplaire n'est pas emprunté, sa date de retour est null\item La caution et l'ISBN d'un livre, les numéros de travée, étagère et niveau sont des entiers positifs
\end{itemize}Invariants à ajouter explicitement dans la documentation, à prendre en compte dans l'implantation pour respecter le cahier des charges
 \end{frame}

 \text{}Attributs
\begin{itemize}\item  Valeur des attributs = état de l'objet Objets différents (identités différentes) peuvent avoir les mêmes attributs Opérations\item Caractéristique partagée par tous les objets de la classe\item  Associe à chaque objet une valeur\item Type associé simple (int, bool...), primitif (Date) ou énuméré
\item Valeur des attributs = état de l'objet
\end{itemize}
Objets différents (identités différentes) peuvent avoir les mêmes attributs
\\
Opérations
\begin{itemize}
\item Service qui peut être demandé à tout objet de la classe\item  Comportement commun à tous les objets de la classe
\end{itemize}
Ne pas confondre avec une méthode = implantation de l'opération

 \begin{frame}
\frametitle{Agrégation} Association particulière entre classes
\begin{itemize}\item  Dissymétrique : une classe prédominante sur l'autre\item  Relation de type composant-composite
\end{itemize}Deux types d'agrégation
\begin{itemize}\item Agrégation faible 
\item Composition
\end{itemize}ExempleLecteur de contenu audio permettant de créer des listes de lecture
  \includegraphics[scale=0.45]{Agregation}
\end{frame}


 \begin{frame}
\frametitle{Composition} Agrégation par valeur
\begin{itemize}\item  Le composite contient ses composants\item  La création ou destruction du composite entraîne la création oudestruction de ses composants\item  Un objet ne fait partie que d'un composite à la fois
\end{itemize}Exemple
\begin{itemize}\item  Un morceau n'appartient qu'à un album\item  La suppression de l'album entraîne la suppression de tous ses morceaux
\end{itemize}

  \includegraphics[scale=0.45]{Composition}
\end{frame}

 \begin{frame}
\frametitle{Limites du diagramme de classes} Diagramme de classes représente la structure du système en termes d'objets et de relations entre ces objetsNe permet pas de représenter :\begin{itemize}\item Valeurs autorisées des attributs\item  Conditions sur les associations\item  Relations entre les attributs ou entre les associations
\end{itemize}Expression des contraintes liées au diagramme :
\begin{itemize}\item  Notes dans le diagramme\item  Texte accompagnant le diagramme\item  OCL : langage de contraintes formel associé à UML
\end{itemize}
\end{frame}




 \begin{frame}
\frametitle{Contraintes, invariants} Propriétés :\begin{itemize}\item   Portant sur les éléments du modèle\item Doivent être vérifiées à tout instant\item  En général, restriction sur les diagrammes d'objets possibles àpartir du diagramme de classes\item  Héritage des contraintes de la super-classe vers les sous-classes
\end{itemize}Contraintes présentes dans le diagramme :
\begin{itemize}\item  Type des attributs\item  Multiplicités des associations
\end{itemize}
\end{frame}

 \begin{frame}
\frametitle{Objectifs de la conception} Objectifs\begin{itemize}\item   Préparer l'implémentation du diagramme de classes et des opérations\item  Avoir une vision d'ensemble avant de programmer
\end{itemize}Intérêt de la conception abstraite, à base de modèles
\begin{itemize}\item Échapper aux détails et contraintes des langages de programmation\item  Vue haut niveau permettant de factoriser les comportements\item Documentation lisible et à part du code
\end{itemize}
\end{frame}

 \begin{frame}
\frametitle{Objectifs de la conception} Préciser et compléter le diagramme de classes\begin{itemize}\item  Préciser les attributs et les associations (visibilité, attributs et associations dérivés, implémentation des associations...)\item Ajouter les classes de contrôle et d'interface
\end{itemize}Décrire les opérations complexes
\begin{itemize}\item  Par de pseudo-code ou des diagrammes de séquence\item  Identifier les méthodes auxiliaires nécessaires\item  Identifier le profil précis des méthodes (paramètres, exceptions)\item  Répartir les méthodes dans les classes\item  Choisir les algorithmes pour implémenter les opérations
\end{itemize}Réfléchir à la réalisation des contraintes et des exigences non fonctionnelles (unicité d'un attribut, contraintes de sécurité, de performance...)
\end{frame}


 \begin{frame}
\frametitle{Diagrammes de séquence (conception)} Objectif : Représenter les communications avec et au sein du logiciel\begin{itemize}\item   Représentation temporelle des interactions entre les objets\item  Chronologie des messages échangés entre les objets et avec les acteurs
\end{itemize}En conception : Décrire la réalisation des cas d'utilisation sur le système décrit par le diagramme de classes
\begin{itemize}\item  Point de vue interne sur le fonctionnement du système\item  Description au niveau de l'instance (état du système à un instant)\item  Description de scénarios particuliers\item  Représentation des échanges de messages\item  Entre les acteurs et le système, entre les objets du système\item  De façon chronologique \end{itemize}
\end{frame}



 \begin{frame}
\frametitle{Diagrammes de séquence (conception)} Éléments du diagramme de séquence\begin{itemize}\item   Acteurs\item  Objets (instances)\item  Messages (cas d'utilisation, appels d?opération)
 \end{itemize}Principes de base : Représentation graphique de la chronologie des échanges de messages avec le système ou au sein du système
 \begin{itemize}\item  " Vie" de chaque entité représentée verticalement\item  Échanges de messages représentés horizontalement
 \end{itemize}
\end{frame}
