 \begin{frame}
Diagramme de flux de données (DFD) :
  \begin{itemize}    \item  La modélisation des fonctions ;   \item  Description des transformations des données par les fonctions ;   \item  Le document contenant la description de ces fonctions s 'intitule "Spécification du processus".
  \end{itemize}  Diagramme de flux du contrôle (CFD) :

 \begin{frame}
Diagramme des transitions d 'états (STD)
  \begin{itemize}    \item   La modélisation comportementale ;   \item   Comportements du logiciel face aux événements extérieurs.   \item   Description des transitions du logiciel d 'un état à l 'autre.   \item   Le document comportant les informations sur le contrôle s 'appelle « Spécification du contrôle »
  \end{itemize}  
\end{frame} 



\begin{frame}
\frametitle{DFD (Data Flow Diagram) }
   \begin{itemize} 
     \item L'une des méthodes disponibles pour analyser les systèmes orientées données.    \item Le DFD met l'emphase sur la logique du système étudié.    \item  Les analystes systèmes peuvent représenter graphiquement les mouvement de données à l'intérieur de l'organisation    \item  Le DFD est une représentation populaire dans la représentation et la modélisation des flux de données.
   \end{itemize}   
\end{frame} 


\begin{frame}
\frametitle{DFD (Data Flow Diagram) }
   \begin{itemize} 
    \item  DFD permet la spécification des fonctions.    \item  DFD utilise deux niveaux de spécifications : 
      \begin{itemize} 
   \item Abstrait : les fonctions globales du logiciel   \item Détail : les sous-fonctions du logiciel en détails.
   \end{itemize}     \item  Ce raffinement est important car réduit la complexité des fonctions.
  \end{itemize}   
 % éléments graphiques
    
  % le traitement ou process = cercle% le flot de données = trait% l'unité de stockage = 2 traits% l'entité externe ou terminateur = rectangle    
\end{frame} 


\begin{frame}
\frametitle{Avantage du DFD}
   \begin{itemize}    \item  Par rapport à une narration des mouvements de données   \item Compréhension des interrelations entre les systèmes et les sous systèmes   \item Documenter ce que système actuel connait   \item Analyser le système proposé
 \end{itemize}  
\end{frame} 

\begin{frame}
\frametitle{DFD vs le cas d'utilisation}
   \begin{itemize}    \item   Le cas d'utilisation décrit les interactions entre le système et un acteur (généralement)   \item   Le cas d'utilisation n'explique pas d'où viennent les données   \item   Par contre le CU fournit de précieuses informations pour créer le DFD
 \end{itemize}  
\end{frame} 

\begin{frame}
\frametitle{Limite du DFD}
   \begin{itemize}    \item   Le DFD ne représente pas les décisions ou les évènements dans le temps.   \item   La fonction du DFD est d'illustrer les sources de données, les destinations, les flux, les magasins de données et les transformations faites aux données.
   \item  Si vous devez modéliser un système qui servira d'entrepôt de données, le DFD ne vous sera pas très utile.   \item   Peu ou pas de processus à documenter 
   \item  Dans ce cas, un ERD risque d'être plus intéressant.   \item   In contrario, le ERD n'illustre pas les transformations que l'on retrouve normalement dans un processus
 \end{itemize}  
\end{frame}


\begin{frame}
\frametitle{Le niveau du contexte}
   \begin{itemize} 
   \item   Contient uniquement un processus représentant l'ensemble du système.   \item  On donne à ce processus le numéro zéro. Ne pas confondre avec le niveau 0.   \item  Tous les agents externes sont représentés dans le diagramme de contexte, ainsi que les flots de données majeurs.   \item   Le diagramme de contexte ne contient aucun magasin de données.
 \end{itemize}  
\end{frame}


\begin{frame}
\frametitle{Les étapes pour créer le DFD}
   \begin{itemize}    \item  Créer un DFD est un travail itératif avec des améliorations continues   \item  Les étapes
       \begin{enumerate}     \item  Créer un diagramme contextuel   \item  Identifier les Cas d'utilisation, là où les utilisateurs utilisent le système   \item  Créer un fragment de DFD pour chaque cas d'utilisation   \item  Créer un diagramme de niveau 0 pour chaque fragment   \item  Décomposer au niveau 1, 2, ...   \item  Revenir à l'étape 1 et réviser si nécessaire   \item  Valider le DFD avec les utilisateurs
       \end{enumerate}  
 \end{itemize}  
\end{frame}

\frametitle{CFD (Control Flow Diagram)}
   \begin{itemize}

      \end{itemize}
